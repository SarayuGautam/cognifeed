"use strict"

const {URL} = require("url")
/** 
 * An abstraction of a web URL
 * @typedef {Object} Link
 * @property {string} baseURL - Holds the base url or the domain
 * @property {string} path - Holds the path of the url, i.e. the part after /
 */

/**
 * Link module
 * @module utils/Link
 */
/**
 * An abstraction of a link
 * @class
 * @param {string} baseURL - The base URL of the link
 * @param {string} path - The path of the link
 * @returns {Link} - A new Link object
 */
module.exports = class Link {
  /**
   * Resolve the current link object into a link URL string
   * @returns {string} - A link href string 
   */
  resolve() {
    return new URL(this.path, this.baseURL).href
  }

  /**
   * Create a new Link Object
   */
  constructor(baseURL, path = "/") {
    /** The base URL of the link */
    this.baseURL = baseURL
    /** The path of the link */
    this.path = path
  }
}
